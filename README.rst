*************
strg.relatime
*************

Human-readable relative times.

Quickstart
==========

.. code-block:: html

    <!-- Dependencies -->
    <script src="score.init.js"></script>
    <script src="score.dom.js"></script>
    <script src="score.relatime.js"></script>
    <script src="strg.relatime.js"></script>

    <!-- HTML -->
    <span id="foo" data-timestamp="1482241653">
        2016-12-20 14:47
    </span>

    <!-- Javascript -->
    <script>
        strg.relatime('#foo');
    </script>
